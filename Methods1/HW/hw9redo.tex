\subsection{Build a linear regression model.}
\label{sec-1-2}


\begin{align*}
\text{Let } \\
V &= \mathrm{volume}\\
D &= \mathrm{diameter}\\
H &= \mathrm{height}\\
\intertext{Assuming that the tree is cylindrical:}\\
V &= ^1/_4 \pi D^2 H\\
\ln V &= \beta_0 + \beta_1 \ln H + \beta_2 \ln D + \epsilon
\end{align*}
